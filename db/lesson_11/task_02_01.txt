-- Практическое задание по теме “NoSQL”
-- 1. В базе данных Redis подберите коллекцию для подсчета посещений с определенных IP-адресов.

-- Создадим множество с уникальными ip-адресами
> SADD ip 'ip1' 'ip2' 'ip3' '...'
-- просмотрим список уникальных ip
> SMEMBERS ip
-- кол-во адресов в коллекции
> SCARD ip

-- Кол-во посещений для каждого ip
> SET ip1 0
> SET ip2 0
> SET ip3 0

-- увеличение кол-ва посещений
> INCR ip1
> GET ip1

> INCR ip2
> GET ip2

> INCR ip2
> GET ip2
